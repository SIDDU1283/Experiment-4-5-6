//Swapping of 2 numbers using 3rd Variable.
#include<stdio.h>
int main()
{
int a=10,b=20,c;
printf("Before Swapping A=%d B=%d",a,b);
c=a;
a=b;
b=c;
printf("After Swapping A=%d B=%d",a,b);
}
//Swapping of 2 numbers without using 3rd Variable.

#include<stdio.h>
int main()
{
int a=10,b=20,c;
printf("Before Swapping A=%d B=%d",a,b);
a=a+b;
b=a-b;
a=a-b;
printf("After Swapping A=%d B=%d",a,b);
}
